<script>                         
    class Vehiculo { // Superclase

        // Los atributos se definene en el contructor
        // En JavaScript, no es necesario especificar la visibilidad (por defecto es público)
        constructor(color) {
            this.color = color;
            this.ruedas = 4;
            this.motor = ""; 
        }

        // Métodos de objeto
        arrancar() {
            console.log("Arranca");
        }

        tipoMotor(motor) {
            this.motor = motor;
            console.log("El motor del coche es " + this.motor);
        }

        getColor() {
            return this.color;
        }

        getRuedas() {
            return this.ruedas;
        }

        // Métodos de clase
        static setTipoVehiculo(tipoVehiculo) { // El atributo de clase se define con el metodo
            Vehiculo.tipoVehiculo = tipoVehiculo; 
        }

        static getTipoVehiculo() {
            return Vehiculo.tipoVehiculo;
        }
    }

    class Coche extends Vehiculo {

        // Constructor
        constructor(color) {
            super(color);
        }

        // Métodos de objeto
        arrancar_hijo() {
            super.arrancar(); // Llama al método del padre
            console.log("El motor ruge como el de un coche");
        }

        // Sobreescritura del método tipoMotor
        tipoMotor(motorCoche) {
            this.motor = motorCoche; // Ten en cuenta que "motor" es un atributo del padre
            console.log("El motor del coche es " + this.motor);
        }
    }
</script>

        // code

    mi_funcion_con_parametros("ejemplo_parametro", "ejemplo_parametro_2");

    // Llamada a función con paso de parámetro por defecto
    // Recuerda que el ámbito local de la variable está aislado del exterior (a diferencia de JavaScript)
    function mi_funcion_default(parametro1, parametro2 = true) {
        
        parametro1 = parametro1.toLowerCase();

        if (parametro2) {
            return parametro1.charAt(0).toUpperCase() + parametro1.slice(1);
        } else {
            return "has modificado el parámetro por defecto";
        }
    }
    mi_funcion_default("ejemplo_parametro");
    mi_funcion_default("ejemplo_parametro", false);

    // Llamada a función con paso de parámetro por valor
    function incrementa(param) {
        param++;
        return param;
    }
    let num = 5;
    console.log(incrementa(num)); // 6
    console.log(num); // 5
