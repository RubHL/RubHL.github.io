<?php
    class Vehiculo{ //Superclase 

        // Atributos/Campos de Objeto
        private $color;
        protected $ruedas;
        var $motor // var es public por defecto
        // Atributos/Campos de Clase
        private static $tipo_vehicuo = "Terrestre" 

        // Metodo constructor (estado inicial al instanciar)
        // Si no existe metodo contructor, se crea uno vacio por defecto
        // $this hace referencia al objeto instanciado ACTUAL
        public function __construct($color){
            $this->ruedas=4;
            $this->color=$color; 
        }

        // Metodos/funciones de Objeto
        function arrancar(){
            echo "Arranca <br>"
        }

        funcion tipo_motor($motor){
            $this->motor=$motor;
            echo "El motor del coche es " . $this->motor . "<br>";
        }

        function get_color(){
            return $this->color;
        }

        function get_ruedas(){
            return $this->ruedas;
        }

        // Metodos/funciones de Clase
        // Si queremos modificar un atributo de clase hay que hacerlo con un metodo de clase
        public static function set_tipo_vehicuo($tipo_vehicuo){ 
            self::$tipo_vehicuo = $tipo_vehicuo;
        }

        // Si queremos obtener un atributo de clase podemos hacerlo a traves del objeto o a traves de su clase
        public static function get_tipo_vehicuo(){ 
            return self::$tipo_vehicuo;
        }

    }

    //::::::::::::::::::::::::::::::::::::::::::::::
                                                            
    class Coche extends Vehiculo{ // Subclase que hereda
        
        // Metodo constructor 
        public function __construct($color){
            $this->ruedas=4;
            $this->color=$color;
        }

        // Metodos/funciones de Objeto
        function arrancar_hijo(){
            parent::arrancar(); // Metodo del padre
            echo "El motor ruge como el de un coche <br>"
        }

        // Realizamos una sobreescritura de metodos de la clase padre pero con $motor_coche
        funcion tipo_motor($motor_coche){
            $this->motor=$motor_coche; // Ten en cuenta que motor es un atributo del padre
            echo "El motor del coche es " . $this->motor . "<br>";
        }
    }
