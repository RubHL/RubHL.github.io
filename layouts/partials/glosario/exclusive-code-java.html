class Vehiculo { // Superclase

    // Atributos de objeto
    private String color;
    protected int ruedas;
    String motor; //Si no se especifica es público por defecto 

    // Atributos de clase
    private static String tipoVehiculo = "Terrestre";

    // Constructor
    public Vehiculo(String color) {
        this.ruedas = 4;
        this.color = color;
    }

    // Métodos de objeto
    void arrancar() {
        System.out.println("Arranca\n");
    }

    void tipoMotor(String motor) {
        this.motor = motor;
        System.out.println("El motor del coche es " + this.motor + "\n");
    }

    String getColor() {
        return this.color;
    }

    int getRuedas() {
        return this.ruedas;
    }

    // Métodos de clase
    public static void setTipoVehiculo(String tipoVehiculo) {
        Vehiculo.tipoVehiculo = tipoVehiculo;
    }

    public static String getTipoVehiculo() {
        return tipoVehiculo;
    }
}

class Coche extends Vehiculo { // Subclase que hereda

    // Constructor
    public Coche(String color) {
        super(color);
    }

    // Métodos de objeto
    void arrancar_hijo() {
        super.arrancar(); // Metodo del padre
        System.out.println("El motor ruge como el de un coche\n");
    }

    // Sobreescritura del método tipoMotor
    @Override
    void tipoMotor(String motorCoche) {
        this.motor = motorCoche; // Ten en cuenta que "motor" es un atributo del padre
        System.out.println("El motor del coche es " + this.motor + "\n");
    }
}
