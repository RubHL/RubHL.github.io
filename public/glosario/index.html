class Vehiculo { // Superclase

    // Atributos de objeto
    private String color;
    protected int ruedas;
    String motor; //Si no se especifica es público por defecto 

    // Atributos de clase
    private static String tipoVehiculo = "Terrestre";

    // Constructor
    public Vehiculo(String color) {
        this.ruedas = 4;
        this.color = color;
    }

    // Métodos de objeto
    void arrancar() {
        System.out.println("Arranca\n");
    }

    void tipoMotor(String motor) {
        this.motor = motor;
        System.out.println("El motor del coche es " + this.motor + "\n");
    }

    String getColor() {
        return this.color;
    }

    int getRuedas() {
        return this.ruedas;
    }

    // Métodos de clase
    public static void setTipoVehiculo(String tipoVehiculo) {
        Vehiculo.tipoVehiculo = tipoVehiculo;
    }

    public static String getTipoVehiculo() {
        return tipoVehiculo;
    }
}

    class Vehiculo{ //Superclase 

        // Atributos/Campos de Objeto
        private $color;

        var $motor // var es public por defecto
        // Atributos/Campos de Clase
        private static $tipo_vehicuo = "Terrestre" 

        // Metodo constructor (estado inicial al instanciar)
        // Si no existe metodo contructor, se crea uno vacio por defecto
        // $this hace referencia al objeto instanciado ACTUAL
        public function __construct($color){
            $this->ruedas=4;
            $this->color=$color; 
        }

        // Metodos/funciones de Objeto
        function arrancar(){
            echo "Arranca <br>"
        }

        funcion tipo_motor($motor){
            $this->motor=$motor;
            echo "El motor del coche es " . $this->motor . "<br>";
        }

        function get_color(){
            return $this->color;
        }

        function get_ruedas(){
            return $this->ruedas;
        }

        // Metodos/funciones de Clase
        // Si queremos modificar un atributo de clase hay que hacerlo con un metodo de clase
        public static function set_tipo_vehicuo($tipo_vehicuo){ 
            self::$tipo_vehicuo = $tipo_vehicuo;
        }

        // Si queremos obtener un atributo de clase podemos hacerlo a traves del objeto o a traves de su clase
        public static function get_tipo_vehicuo(){ 
            return self::$tipo_vehicuo;
        }

    }
